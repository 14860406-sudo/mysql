Explicitly define table schema with primary key using SQLAlchemy Core column_definitions = [] # Add the primary key column explicitly column_definitions.append(Column('id', BigInteger, primary_key=True)) # Map other DataFrame columns to SQLAlchemy types for col_name, dtype in df.dtypes.items(): if col_name == 'id': continue # 'id' is already defined with PK # Simple type mapping based on pandas dtypes if pd_types.is_integer_dtype(dtype): column_definitions.append(Column(col_name, BigInteger)) elif pd_types.is_float_dtype(dtype): column_definitions.append(Column(col_name, Float)) elif pd_types.is_datetime64_any_dtype(dtype): column_definitions.append(Column(col_name, DateTime)) else: # Catch-all for strings, objects, etc. Default to String(255) # Consider using Text for potentially longer strings, or infer length dynamically column_definitions.append(Column(col_name, String(255))) # Define the table object my_table = Table(table_name, metadata, *column_definitions) # Drop existing table if it exists (to simulate 'if_exists='replace' behavior) with engine.connect() as connection: if engine.dialect.has_table(connection, table_name): print(f"Dropping existing table '{table_name}'...") my_table.drop(connection) print(f"Table '{table_name}' dropped.") # Create the table in the database print(f"Creating table '{table_name}' with primary key...") my_table.create(connection) print(f"Table '{table_name}' created successfully.") # 6. import pandas as pd from sqlalchemy import create_engine import matplotlib.pyplot as plt import seaborn as sns # 1. 建立連線 (請換成你的密碼) USER = 'avnadmin' PASSWORD = 'AVNS_Ggqii1pHOaZqTR0hJjC' HOST = 'mysql-23f76110-me-1dbd.d.aivencloud.com' PORT = '23367' DB_NAME = 'defaultdb' # 或 shop_data，視你上傳的位置而定 connection_uri = f"mysql+pymysql://{USER}:{PASSWORD}@{HOST}:{PORT}/{DB_NAME}?ssl_verify_cert=false" engine = create_engine(connection_uri) # 2. 抓取資料庫內的資料 # 假設你的資料表名稱是 excel_upload_test query = "SELECT * FROM excel_upload_test" df = pd.read_sql(query, engine) # 設定中文字體 (Colab 環境) plt.rcParams['font.sans-serif'] = ['Liberation Sans'] plt.rcParams['axes.unicode_minus'] = False # --- 開始繪圖 --- # 建立一個大畫布，容納多個圖表 plt.figure(figsize=(15, 10)) # 圖表 1：長條圖 (Bar Chart) - 範例：不同產品類別的數量 plt.subplot(2, 2, 1) # 假設你的 Excel 裡有一欄叫 'category' (請根據你的欄位名稱修改) # 如果不知道欄位名，可以先用 sns.countplot(data=df.iloc[:, 1:2]) 測試 sns.countplot(x=df.columns[2], data=df) plt.title('Product Category Count') plt.xticks(rotation=45) # 圖表 2：折線圖 (Line Chart) - 範例：銷售金額趨勢 plt.subplot(2, 2, 2) # 假設你有 'date' 和 'net_sales' 欄位 # sns.lineplot(x='date', y='net_sales', data=df) plt.plot(df.index, df['net_sales'] if 'net_sales' in df.columns else df.iloc[:, -1]) plt.title('Sales Trend') # 圖表 3：散佈圖 (Scatter Plot) - 範例：折扣與銷售額的關係 plt.subplot(2, 2, 3) if 'discount_amount' in df.columns and 'net_sales' in df.columns: sns.scatterplot(x='discount_amount', y='net_sales', data=df) plt.title('Discount vs Sales') # 圖表 4：圓餅圖 (Pie Chart) - 範例：客戶比例 plt.subplot(2, 2, 4) df.iloc[:, 3].value_counts().plot.pie(autopct='%1.1f%%') plt.title('Segment Distribution') plt.tight_layout() plt.show()
